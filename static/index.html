// Auto-generated · BTC/USD 4H
//@version=5
strategy("dotverse FCR+FVG", overlay=true, initial_capital=10000,
         default_qty_type=strategy.percent_of_equity, default_qty_value=2)

// ─── Inputs ───
atrLength  = input.int(14, "ATR Length")
emaFast    = input.int(20, "EMA Fast")
emaSlow    = input.int(50, "EMA Slow")
rsiLen     = input.int(14, "RSI Length")
rsiOB      = input.float(65.0, "RSI Max Entry")

// ─── Calculations ───
atr    = ta.atr(atrLength)
ema20  = ta.ema(close, emaFast)
ema50  = ta.ema(close, emaSlow)
rsi    = ta.rsi(close, rsiLen)

// ─── FVG Detection ───
bullFVG   = low[0] > high[2]
bearFVG   = high[0] < low[2]

// ─── Trend Filter ───
bullTrend = close > ema20 and ema20 > ema50
bearTrend = close < ema20 and ema20 < ema50

// ─── Entry Conditions ───
longCondition  = bullTrend and bullFVG and rsi < rsiOB
shortCondition = bearTrend and bearFVG and rsi > (100 - rsiOB)

if longCondition
    sl  = close - atr * 1.5
    tp1 = close + atr * 2.5
    tp2 = close + atr * 4.0
    tp3 = close + atr * 6.0
    strategy.entry("Long", strategy.long)
    strategy.exit("L TP1", "Long", qty_percent=50, stop=sl, limit=tp1)
    strategy.exit("L TP2", "Long", qty_percent=30, stop=sl, limit=tp2)
    strategy.exit("L TP3", "Long", qty_percent=20, stop=sl, limit=tp3)
